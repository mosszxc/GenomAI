## 🧮 4. Алгоритмы расчета

### 4.1 Расчет теплопотерь

```python
def calculate_heat_loss(area, height, building_type, region_temp):
    """
    Расчет теплопотерь здания
    
    Параметры:
    - area: площадь здания (м²)
    - height: высота потолков (м)
    - building_type: тип здания (жилое, производственное)
    - region_temp: средняя температура региона зимой (°C)
    """
    # Объем здания
    volume = area * height
    
    # Коэффициент теплопотерь (зависит от типа здания)
    heat_loss_coefficient = {
        'residential': 0.8,  # Вт/(м³·°C)
        'industrial': 1.2,
        'office': 1.0
    }
    
    # Температура внутри (обычно 20°C)
    inside_temp = 20
    
    # Разница температур
    delta_temp = inside_temp - region_temp
    
    # Теплопотери (Вт)
    heat_loss = volume * heat_loss_coefficient[building_type] * delta_temp
    
    return heat_loss
```

### 4.2 Расчет потребности в угле

```python
def calculate_coal_need(heat_loss, heating_days, boiler_efficiency, coal_heat_value):
    """
    Расчет необходимого количества угля
    
    Параметры:
    - heat_loss: теплопотери (Вт)
    - heating_days: количество дней отопительного сезона
    - boiler_efficiency: КПД котла (0.7-0.9)
    - coal_heat_value: теплотворная способность угля (кДж/кг)
    """
    # Теплопотери за сезон (кДж)
    total_heat_loss = heat_loss * heating_days * 24 * 3600 / 1000
    
    # Учет КПД котла
    required_energy = total_heat_loss / boiler_efficiency
    
    # Необходимое количество угля (кг)
    coal_kg = required_energy / coal_heat_value
    
    # Переводим в тонны
    coal_tons = coal_kg / 1000
    
    # Запас 10-15%
    coal_tons_with_reserve = coal_tons * 1.15
    
    return coal_tons_with_reserve
```

## 📊 6. Схема базы данных

```sql
-- Расчеты
CREATE TABLE calculations (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES auth.users(id),
    building_area DECIMAL(10, 2),
    building_height DECIMAL(5, 2),
    building_type TEXT,
    region TEXT,
    coal_type TEXT,
    coal_amount DECIMAL(10, 3), -- тонны
    price_per_ton DECIMAL(10, 2),
    total_cost DECIMAL(12, 2),
    created_at TIMESTAMP DEFAULT now()
);

-- История закупок
CREATE TABLE purchases (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    calculation_id UUID REFERENCES calculations(id),
    purchase_date DATE,
    amount DECIMAL(10, 3),
    price_per_ton DECIMAL(10, 2),
    supplier TEXT,
    notes TEXT,
    created_at TIMESTAMP DEFAULT now()
);

-- Настройки пользователя
CREATE TABLE user_settings (
    user_id UUID PRIMARY KEY REFERENCES auth.users(id),
    default_region TEXT,
    default_coal_type TEXT,
    currency TEXT DEFAULT 'RUB',
    created_at TIMESTAMP DEFAULT now(),
    updated_at TIMESTAMP DEFAULT now()
);
```
